## 目录说明

```
├── public
│   └── favicon.ico          # 标签页图标
|   └── index.html           # Vue 入口模板
|   └── wwLogin-1.2.4.js     # 企微扫码登录生成二维码插件
├── src
│   ├── api                  # Api ajax
│   ├── assets               # 本地静态资源
│   ├── components           # 业务通用组件
│   ├── layout           	 # 系统布局文件
│   ├── permission           # 权限控制，包括路由控制 以及 按钮控制
│   ├── router               # Vue-Router 路由配置
│   ├── store                # Vuex 单一状态管理
│   ├── utils                # 工具库
│   ├── views                # 业务页面入口和常用模板
│   ├── App.vue              # Vue 模板入口
│   └── main.js              # Vue 入口 JS
│   └── ant-design.config.js # ant-design-vue 组件引入配置
│   └── global.less          # 全局样式
├── README.md
└── package.json
```
